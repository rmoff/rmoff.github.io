SELECT filename, CAST("Physical Postal Code" AS VARCHAR) AS "Physical Postal Code"
  FROM read_parquet('s3://drones03/main/drone-registrations/Registations-P107-Active-2016.parquet', filename=true)
USING SAMPLE 5 ROWS
