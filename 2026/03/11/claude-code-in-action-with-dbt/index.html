stg_stations.sql 
with source as (

    select * from {{ source('flood_monitoring_api', 'raw_stations') }}

),

renamed as (

    select
        -- Primary key
        notation                                        as station_id,

        -- Descriptive attributes
        label                                           as station_name,
        town,
        "riverName"                                     as river_name,
        "catchmentName"                                 as catchment_name,

        -- Geographic coordinates (may be null for some stations)
        try_cast(lat as double)                         as latitude,
        try_cast(long as double)                        as longitude,
        try_cast(easting as integer)                    as easting,
        try_cast(northing as integer)                   as northing,

        -- Station metadata
        "RLOIid"                                        as rloi_id,
        "stationReference"                              as station_reference,
        "wiskiID"                                       as wiski_id,
        "dateOpened"                                    as date_opened,

        -- Status: extract the short status from the URI
        -- Handles: JSON strings, arrays (take first element), nulls, and
        -- the "ukcmf" variant
        case
            when status is null then 'Unknown'
            else regexp_replace(
                -- For arrays, take the first element; for strings, use as-is
                case
                    when json_type(status) = 'ARRAY'
                        then cast(status->>0 as varchar)
                    else cast(status as varchar)
                end,
                '.*status',
                ''
            )
        end                                             as status,

        -- Audit
        loaded_at

    from source

)

select * from renamed

with source as (

    select * from {{ source('flood_monitoring_api', 'raw_readings') }}

),

cleaned as (

    select
        -- Timestamp
        cast("dateTime" as timestamp)                   as reading_timestamp,

        -- Foreign key: extract measure notation from the URI
        regexp_replace(
            measure,
            'http://environment\.data\.gov\.uk/flood-monitoring/id/measures/',
            ''
        )                                               as measure_id,

        -- Value cleaning: handle pipe-delimited values (e.g. "0.770|0.688")
        -- by taking the first value, then cast to double
        try_cast(
            case
                when value like '%|%' then split_part(value, '|', 1)
                else value
            end as double
        )                                               as reading_value,

        -- Flag rows that had pipe-delimited values
        value like '%|%'                                as has_pipe_delimited_value,

        -- Keep original value for debugging
        value                                           as raw_value,

        -- Audit
        loaded_at

    from source

),

filtered as (

    -- Exclude rows where value is null or NaN after cleaning
    select * from cleaned
    where reading_value is not null
      and isnan(reading_value) = false

)

select * from filtered
